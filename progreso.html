analizar_consistencia <- function(datos, dimension) {
  
  cat("\n", rep("=", 20), "ANÁLISIS DIMENSIÓN:", dimension, rep("=", 20), "\n")
  
  # A. Filtrado y transformación a formato ancho (Matriz para PCA/Correlación)
  df_wide <- datos |> 
    filter(dimension_id == dimension, year == 2022, !is.na(value)) |> 
    select(geo_id, indicator_id, value) |> 
    pivot_wider(names_from = indicator_id, values_from = value) |> 
    column_to_rownames(var = "geo_id")
  
  # B. Limpieza: Eliminar indicadores que no varían (SD = 0) o tienen NAs
  # Elimino esos porque el análisis de correlación falla si la desviación estándar es 0
  df_wide <- df_wide[, sapply(df_wide, sd, na.rm = TRUE) > 0]
  
  # C. Verificación de viabilidad
  if (ncol(df_wide) < 2) {
    cat("AVISO: No hay suficientes indicadores con variación en la dimensión", dimension, "\n")
    return(NULL)
  }
  
  # D. Matriz de Correlación (Figura 1)
  cor_mat <- cor(df_wide, use = "pairwise.complete.obs")
  
  p <- ggcorrplot(cor_mat, 
                  hc.order = TRUE, 
                  type = "lower",
                  lab = TRUE, 
                  lab_size = 3,
                  title = paste("Matriz de Correlación:", dimension),
                  colors = c("#E4672E", "white", "#6D9EC1"))
  print(p)
  
  # E. Test de Bartlett y KMO (Metodología de Lüdecke et al., 2021)
  # uso check_factorstructure() del paquete performance
  cat("\n--- Test de Adecuación (Bartlett & KMO) ---\n")
  print(check_factorstructure(df_wide))
  
  # F. Alpha de Cronbach (Consistencia Interna)
  # check.keys = TRUE invierte automáticamente indicadores con carga negativa
  cat("\n--- Consistencia Interna (Alpha de Cronbach) ---\n")
  alpha_res <- psych::alpha(df_wide, check.keys = TRUE)
  cat("Alpha de Cronbach estandarizado:", round(alpha_res$total$std.alpha, 3), "\n")
}

df_clean <- df  |>

# 2. PCA POR DIMENSIÓN
pca_results_list <- list() 
dimension_scores_list <- list()

for (dim in sort(unique(df_clean$dimension_id))) {
  
  # Seleccionamos solo lo necesario para el PCA (quitamos dimension_id aquí)
  pca_data <- df_clean |>
    filter(dimension_id == dim) |>
    select(geo_id, indicator_id, value) |> # <--- EL FILTRO CRÍTICO
    pivot_wider(names_from = indicator_id, values_from = value) |>
    column_to_rownames(var = "geo_id")
  
  # Limpieza: quitar indicadores con >50% NAs e imputar con la media
  pca_data_clean <- pca_data |>
    select(where(~mean(is.na(.)) <= 0.5)) |>
    mutate(across(everything(), ~ifelse(is.na(.), mean(., na.rm = TRUE), .)))
  
  if (ncol(pca_data_clean) >= 2) {
    pca_obj <- PCA(pca_data_clean, scale.unit = TRUE, ncp = 1, graph = FALSE)
    pca_results_list[[dim]] <- pca_obj
    
    # Guardamos los scores del PC1
    scores <- as.data.frame(pca_obj$ind$coord)
    colnames(scores) <- paste0(dim, "_PC1")
    scores$region <- rownames(scores)
    dimension_scores_list[[dim]] <- scores
  }
}

# 3. AGREGACIÓN FINAL (ÍNDICE SINTÉTICO)
# Unimos los 3 PC1 (ECO, ENV, SOC) en una tabla
final_input_data <- Reduce(function(x, y) full_join(x, y, by = "region"), dimension_scores_list) |>
  column_to_rownames(var = "region") |>
  drop_na()

# PCA Final (Dimensión de Dimensiones)

# Este es el más complejo. Para cada país, buscamos los pesos que maximicen su nota.
calculate_bod <- function(country_data, full_data) {
  n_vars <- length(country_data)
  
  # Función objetivo: Maximizar (x1*w1 + x2*w2 + ...) para el país actual
  f.obj <- as.numeric(country_data)
  
  # Restricciones:
  # 1. El índice calculado con estos pesos no puede ser > 1 para NINGÚN país
  f.con <- as.matrix(full_data)
  f.dir <- rep("<=", nrow(full_data))
  f.rhs <- rep(1, nrow(full_data))
  
  # 2. Restricción adicional: Pesos deben ser positivos (>= 0.05 para evitar ceros totales)
  # Esto evita que un país ignore totalmente sus debilidades
  f.con <- rbind(f.con, diag(n_vars))
  f.dir <- c(f.dir, rep(">=", n_vars))
  f.rhs <- c(f.rhs, rep(0.01, n_vars)) # Peso mínimo 1%
  
  # Resolver Programación Lineal
  res <- lp("max", f.obj, f.con, f.dir, f.rhs)
  
  return(res$objval) # Retorna el valor óptimo del índice
}

Métodos OCDE con nuestros datos

Dividimos en dos niveles: * Nivel 1: de indicadores individuales a dimensión. * Nivel 2: de dimensiones a índice final.

Tenemos que decidir qué metodología usar. 1. Podemos ponderar solo en el índice final. Asumimos que, dentro de una dimensión, todos los indicadores indiv son igualmente importantes (es decir, hacemos la media aritmética para calcular índice por dimensión), y luego ejecutamos los 7 métodos de ponderación para ver los pesos de cada una de las 3 dimensiones y obtener el índice final. Esto es más transparente, interpretable, sencillo de comunicar… Pero, si un indicador es irrelevante lo estaríamos contando sin necesidad, o si está muy correlacionado con otro, estaríamos contando lo mismo dos veces (info redundante).

  1. Ponderamos en dos niveles. En el nivel 1 usamos p.ej. PCA para que los indicadores que más varían tengan más peso ; y en el nivel 2 usamos los 7 métodos para decidir cuánto pesa ECO frente a SOC o ENV. Esto sería lo ideal si queremos ser rigurosos, pero es mucho más difícil de explicar. P. ej. si el índice de una región sale bajo, ¿es porque sus indicadores individuales pesan poco o porque una de las 3 dimensiones tiene poco peso?

  2. Ponderamos en dos niveles. 7 métodos en el 1 y 7 métodos en el 2.

# Función auxiliar para limpiar datos (Varianza 0 y NAs) antes de calcular
limpiar_matriz <- function(df_input) {
  mat <- df_input %>%
    select(where(~ sd(., na.rm = TRUE) > 0)) %>% # Quitar columnas constantes
    mutate(across(everything(), ~ ifelse(is.na(.), mean(., na.rm = TRUE), .))) %>% # Imputar media
    as.matrix()
  return(mat)
}

# FUNCIÓN MAESTRA DE PONDERACIÓN (Se usa para Nivel 1 y Nivel 2)
# ==============================================================================
calcular_score <- function(matriz_datos, metodo) {
  
  n_cols <- ncol(matriz_datos)
  n_rows <- nrow(matriz_datos)
  scores <- numeric(n_rows)
  
  if (metodo == "EW") { # Equal Weighting
    scores <- rowMeans(matriz_datos)
    
  } else if (metodo == "PCA") {
    if(n_cols < 2) return(rowMeans(matriz_datos)) # Fallback si solo queda 1 columna
    pca <- prcomp(matriz_datos, scale. = TRUE)
    w <- (pca$rotation[,1]^2) / sum(pca$rotation[,1]^2)
    scores <- matriz_datos %*% w
    
  } else if (metodo == "BOD") { 
    # Benefit of the Doubt (Optimización)
    # NOTA: Fallará conceptualmente con negativos (Z-scores)
    bod_fun <- function(row_vals) {
      f.obj <- as.numeric(row_vals)
      # Restricciones: suma ponderada <= 1 (max teórico z-score normalizado, aquí fallará)
      # y pesos >= 0.01
      f.con <- rbind(matriz_datos, diag(n_cols))
      f.dir <- c(rep("<=", n_rows), rep(">=", n_cols))
      f.rhs <- c(rep(max(matriz_datos)+1, n_rows), rep(0.01, n_cols)) 
      
      # Intentamos resolver. Si falla por negativos, devuelve NA o algo raro
      tryCatch({
        res <- lp("max", f.obj, f.con, f.dir, f.rhs)
        return(res$objval)
      }, error = function(e) return(NA))
    }
    scores <- apply(matriz_datos, 1, bod_fun)
    
  } else if (metodo == "UCM") {
    # Unobserved Components (Correlación)
    cor_m <- cor(matriz_datos)
    diag(cor_m) <- 0
    w <- colMeans(abs(cor_m)) / sum(colMeans(abs(cor_m)))
    scores <- matriz_datos %*% w
    
  } else {
    # Para BAP, AHP, CA (Subjetivos)
    # Como no podemos entrevistar expertos para cada dimensión individualmente ahora,
    # simulamos pesos decrecientes para que el código corra.
    # (El 1er indicador pesa más que el último)
    w_sim <- sort(runif(n_cols), decreasing = TRUE)
    w_sim <- w_sim / sum(w_sim)
    scores <- matriz_datos %*% w_sim
  }
  
  return(as.numeric(scores))
}
